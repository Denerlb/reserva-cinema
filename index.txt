<script>
//muda a cor e soma os valores

const poltronas = document.querySelectorAll(".poltrona");
  const info = document.getElementById("psel");
  const valorDiv = document.getElementById("valor");

  poltronas.forEach(function(poltrona) {
    poltrona.addEventListener("click", function() {
      // Alterna seleção
      this.classList.toggle("selecionada");

      // Coleta poltronas selecionadas
      const selecionadas = Array.from(poltronas)
        .filter(p => p.classList.contains("selecionada"));

      // Atualiza texto com IDs
      const ids = selecionadas.map(p => p.id);
      info.textContent = ids.length > 0
        ? "Poltronas selecionadas: " + ids.join(", ")
        : "Poltronas selecionadas: Nenhuma";

      // Soma os valores
      const total = selecionadas.reduce((sum, p) => {
        return sum + parseFloat(p.dataset.valor);
      }, 0);

      // Atualiza valor total
      valorDiv.textContent = "Valor total: R$ " + total.toFixed(2).replace(".", ",");
    });
  });


//LISTA SUSPENSA

const select = document.getElementById("idmovie");
  const resultado = document.getElementById("nomefilme");

  select.addEventListener("change", function() {
    const nomeSelecionado = this.value;
    resultado.textContent = nomeSelecionado
      ? "Sessão: " + nomeSelecionado
      : "Nome selecionado: Nenhum";
  });

// Seleciona todos os elementos h3 dentro da div "textos"
  const h3Elements = document.querySelectorAll('#info h3');
  const catDiv = document.getElementById('cat');

  h3Elements.forEach(h3 => {
    h3.addEventListener('click', () => {
      catDiv.textContent = h3.textContent;
    });
  });
</script>
